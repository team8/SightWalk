10  GLdouble x, GLdouble y,

12  GLint viewport[4]
13 ) {
14  GLfloat m[16];
15  GLfloat sx, sy;
16  GLfloat tx, ty;
17  sx = viewport[2] / (GLfloat)width;
18  sy = viewport[3] / (GLfloat)height;
19  tx = (viewport[2] + 2.0f * (viewport[0] - (GLfloat)x)) / (GLfloat)width;
20  ty = (viewport[3] + 2.0f * (viewport[1] - (GLfloat)y)) / (GLfloat)height;
21 #define M(row, col) m[col*4+row]
22  M(0, 0) = sx;
23  M(0, 1) = 0.0f;
24  M(0, 2) = 0.0f;
25  M(0, 3) = tx;
26  M(1, 0) = 0.0f;
27  M(1, 1) = sy;

34  M(3, 0) = 0.0f;
35  M(3, 1) = 0.0f;
36  M(3, 2) = 0.0f;
37  M(3, 3) = 1.0f;
