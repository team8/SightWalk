55 struct Marker
56 {
57  enum Direction
58  {
59  Horizontal,
60  Vertical
61  };
62 
63  enum Equality
64  {
65  LessThan = -1,
66  Equal = 0,
67  GreaterThan = 1
68  };
69 
70  Marker(Direction d, float value, Equality leg = Equal, Colour c = Colour() )
71  : colour(c)
72  {
73  if(d == Horizontal) {
74  range.x = Rangef::Open();
75  range.y = Rangef::Containing(value);
76  if(leg == LessThan) {
77  range.y.Insert(std::numeric_limits<float>::lowest() );
78  }else if(leg == GreaterThan) {
79  range.y.Insert(std::numeric_limits<float>::max() );
80  }
81  }else if(d == Vertical) {
82  range.x = Rangef::Containing(value);
83  range.y = Rangef::Open();
84  if(leg == LessThan) {
85  range.x.Insert(std::numeric_limits<float>::lowest() );
86  }else if(leg == GreaterThan) {
87  range.x.Insert(std::numeric_limits<float>::max() );
88  }
89  }
90  }
91 
92  Marker(const XYRangef& range, const Colour& c = Colour() )
93  : range(range), colour(c)
94  {
95  }
96 
97  XYRangef range;
98  Colour colour;
99 };
