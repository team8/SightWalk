96  static PyObject* Py_write(PyPangoIO* self, PyObject *args)
97  {
98  const char *text = 0;
99  if (PyArg_ParseTuple(args, "s", &text)) {
100  self->buffer += std::string(text);
101  size_t nl = self->buffer.find_first_of('\n');
102  while(nl != std::string::npos) {
103  const std::string line = self->buffer.substr(0,nl);
104  self->line_queue.push(ConsoleLine(line,self->line_type));
105  self->buffer = self->buffer.substr(nl+1);
106  nl = self->buffer.find_first_of('\n');
107  }
108  }
109  Py_RETURN_NONE;
110  }
