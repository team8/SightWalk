51 /// Simple statistics recorded for a logged input dimension.
52 struct DimensionStats
53 {
54  DimensionStats()
55  {
56  Reset();
57  }
58 
59  void Reset()
60  {
61  isMonotonic = true;
62  sum = 0.0f;
63  sum_sq = 0.0f;
64  min = std::numeric_limits<float>::max();
65  max = std::numeric_limits<float>::lowest();
66  }
67 
68  void Add(const float v)
69  {
70  isMonotonic = isMonotonic && (v >= max);
71  sum += v;
72  sum_sq += v*v;
73  min = std::min(min, v);
74  max = std::max(max, v);
75  }
76 
77  bool isMonotonic;
78  float sum;
79  float sum_sq;
80  float min;
81  float max;
82 };
