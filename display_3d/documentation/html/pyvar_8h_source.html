40 PyObject* GetPangoVarAsPython(const std::string& name)
41 {
42  VarState::VarStoreContainer::iterator i = VarState::I().vars.find(name);
43  if(i != VarState::I().vars.end()) {
44  VarValueGeneric* var = i->second;
45 
46  try{
47  if( !strcmp(var->TypeId(), typeid(bool).name() ) ) {
48  const bool val = Var<bool>(*var).Get();
49  return PyBool_FromLong( val );
50  }else if( !strcmp(var->TypeId(), typeid(short).name() ) ||
51  !strcmp(var->TypeId(), typeid(int).name() ) ||
52  !strcmp(var->TypeId(), typeid(long).name() ) ) {
53  const long val = Var<long>(*var).Get();
54  return PyLong_FromLong( val );
55  }else if( !strcmp(var->TypeId(), typeid(double).name() ) ||
56  !strcmp(var->TypeId(), typeid(float).name() ) ) {
57  const double val = Var<double>(*var).Get();
58  return PyFloat_FromDouble(val);
59  }else{
60  const std::string val = var->str->Get();
61 #if PY_MAJOR_VERSION >= 3
62  return PyUnicode_FromString(val.c_str());
63 #else
64  return PyString_FromString(val.c_str());
65 #endif
66  }
67  }catch(const std::exception&) {
68  }
69  }
70 
71  Py_RETURN_NONE;
72 }

74 void SetPangoVarFromPython(const std::string& name, PyObject* val)
75 {
76  try{
77  if (PyFloat_Check(val)) {
78  pangolin::Var<double> pango_var(name);
79  pango_var = PyFloat_AsDouble(val);
80  pango_var.Meta().gui_changed = true;
81  }else if (PyLong_Check(val)) {
82  pangolin::Var<long> pango_var(name);
83  pango_var = PyLong_AsLong(val);
84  pango_var.Meta().gui_changed = true;
85  }else if (PyBool_Check(val)) {
86  pangolin::Var<bool> pango_var(name);
87  pango_var = (val == Py_True) ? true : false;
88  pango_var.Meta().gui_changed = true;
89  }
90 #if PY_MAJOR_VERSION >= 3
91  else if (PyUnicode_Check(val)) {
92  pangolin::Var<std::string> pango_var(name);
93  pango_var = PyUnicode_AsUTF8(val);
94  pango_var.Meta().gui_changed = true;
95  }
96 #else
97  else if (PyString_Check(val)) {
98  pangolin::Var<std::string> pango_var(name);
99  pango_var = PyString_AsString(val);
100  pango_var.Meta().gui_changed = true;
101  } else if (PyInt_Check(val)) {
102  pangolin::Var<int> pango_var(name);
103  pango_var = PyInt_AsLong(val);
104  pango_var.Meta().gui_changed = true;
105  }
106 #endif
107  else {
108  PyUniqueObj pystr = PyObject_Repr(val);
109 #if PY_MAJOR_VERSION >= 3
110  const std::string str = PyUnicode_AsUTF8(pystr);
111 #else
112  const std::string str = PyString_AsString(pystr);
113 #endif
114  pangolin::Var<std::string> pango_var(name);
115  pango_var = str;
116  pango_var.Meta().gui_changed = true;
117  }
118  FlagVarChanged();
119  }catch(const std::exception& e) {
120  pango_print_error("%s\n", e.what());
121  }
122 }

161  static PyObject* Py_getattr(PyVar *self, char* name)
162  {
163  const std::string prefix = self->ns + ".";
164  const std::string full_name = self->ns.empty() ? name : prefix + std::string(name);
165 
166  if( !strcmp(name, "__call__") ||
167  !strcmp(name, "__dict__") ||
168  !strcmp(name, "__methods__") ||
169  !strcmp(name, "__class__") )
170  {
171  // Default behaviour
172 #if PY_MAJOR_VERSION >= 3
173  return PyObject_GenericGetAttr((PyObject*)self, PyUnicode_FromString(name));
174 #else
175  return PyObject_GenericGetAttr((PyObject*)self, PyString_FromString(name));
176 #endif
177  } else if( !strcmp(name, "__members__") ) {
178  const int nss = prefix.size();
179  PyObject* l = PyList_New(0);
180  for(const std::string& s : VarState::I().var_adds) {
181  if(!s.compare(0, nss, prefix)) {
182  size_t dot = s.find_first_of('.', nss);
183  std::string val = (dot != std::string::npos) ? s.substr(nss, dot - nss) : s.substr(nss);
184 #if PY_MAJOR_VERSION >= 3
185  PyList_Append(l, PyUnicode_FromString(val.c_str()));
186 #else
187  PyList_Append(l, PyString_FromString(val.c_str()));
188 #endif
189  }
190  }
191 
192  return l;
193  }else if( pangolin::VarState::I().Exists(full_name) ) {
194  return GetPangoVarAsPython(full_name);
195  }else{
196  PyVar* obj = (PyVar*)PyVar::Py_new(&PyVar::Py_type,NULL,NULL);
197  if(obj) {
198  obj->ns = full_name;
199  return PyObject_Init((PyObject *)obj,&PyVar::Py_type);
200  }
201  return (PyObject *)obj;
202  }
203 
204  Py_RETURN_NONE;
205  }
206 
207  static int Py_setattr(PyVar *self, char* name, PyObject* val)
208  {
209  const std::string full_name = self->ns.empty() ? name : self->ns + "." + std::string(name);
210  SetPangoVarFromPython(full_name, val);
211  return 0;
212  }

214  std::string ns;
215 };
