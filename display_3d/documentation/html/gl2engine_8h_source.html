71  GlEngine()
72  {
73  // Initialise default state
74  projection.push(IdentityMatrix());
75  modelview.push(IdentityMatrix());
76  currentmatrix = &modelview;
77 
78  // Set GL_TEXTURE0 as default active texture
79  glActiveTexture(GL_TEXTURE0);
80 
81  // Compile and link shaders
82  prog_fixed.AddShader(GlSlVertexShader, vert);
83  prog_fixed.AddShader(GlSlFragmentShader, frag);
84  prog_fixed.BindPangolinDefaultAttribLocationsAndLink();
85 
86  // Save locations of uniforms
87  u_color = prog_fixed.GetUniformHandle("u_color");
88  u_modelViewMatrix = prog_fixed.GetUniformHandle("u_modelViewMatrix");
89  u_modelViewProjectionMatrix = prog_fixed.GetUniformHandle("u_modelViewProjectionMatrix");
90  u_texture = prog_fixed.GetUniformHandle("u_texture");
91  u_textureEnable = prog_fixed.GetUniformHandle("u_textureEnable");
92 
93  // Initialise default uniform values
94  UpdateMatrices();
95  SetColor(1.0,1.0,1.0,1.0);
96  }
97 
98  void UpdateMatrices()
99  {
100  OpenGlMatrix pmv = projection.top() * modelview.top();
101  prog_fixed.SaveBind();
102  glUniformMatrix4fv( u_modelViewMatrix, 1, false, modelview.top().m );
103  glUniformMatrix4fv( u_modelViewProjectionMatrix, 1, false, pmv.m );
104  prog_fixed.Unbind();
105  }

107  void SetColor(float r, float g, float b, float a)
108  {
109  prog_fixed.SaveBind();
110  glUniform4f( u_color, r, g, b, a);
111  prog_fixed.Unbind();
112  }

114  void EnableTexturing(GLboolean v)
115  {
116  prog_fixed.SaveBind();
117  glUniform1i( u_textureEnable, v);
118  prog_fixed.Unbind();
119  }

183  if(cap == GL_VERTEX_ARRAY) {
184  glEnableVertexAttribArray(pangolin::DEFAULT_LOCATION_POSITION);
185  }else if(cap == GL_COLOR_ARRAY) {
186  glEnableVertexAttribArray(pangolin::DEFAULT_LOCATION_COLOUR);
187  }else if(cap == GL_NORMAL_ARRAY) {
188  glEnableVertexAttribArray(pangolin::DEFAULT_LOCATION_NORMAL);
189  }else if(cap == GL_TEXTURE_COORD_ARRAY) {
190  glEnableVertexAttribArray(pangolin::DEFAULT_LOCATION_TEXCOORD);
191  gl.EnableTexturing(true);
