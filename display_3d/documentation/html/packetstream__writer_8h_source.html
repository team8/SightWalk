154 inline picojson::value SourceStats(const std::vector<PacketStreamSource>& srcs)
155 {
156  picojson::value stat;
157  stat["num_sources"] = srcs.size();
158  stat["src_packet_index"] = picojson::array();
159  stat["src_packet_times"] = picojson::array();
160 
161  for(auto& src : srcs) {
162  picojson::array pkt_index, pkt_times;
163  for (const PacketStreamSource::PacketInfo& frame : src.index) {
164  pkt_index.emplace_back(frame.pos);
165  pkt_times.emplace_back(frame.capture_time);
166  }
167  stat["src_packet_index"].push_back(std::move(pkt_index));
168  stat["src_packet_times"].push_back(std::move(pkt_times));
169  }
170  return stat;
171 }
